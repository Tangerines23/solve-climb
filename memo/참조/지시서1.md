// (v3.0) 로직/상태/유틸은 v2.x와 동일하게 사용

export function MathQuizPage() {

  useEffect(() => {
    generateNewQuestion();
  }, [generateNewQuestion]);

  const handleSubmit = useCallback(

      e.preventDefault();
      // ... (v2.x와 동일한 isSubmitting, 빈 값, NaN, 범위 초과 검사) ...
      // ... (v2.x와 동일한 정답/오답 처리, 후속 처리, 포커스 관리) ...
    },
    [/* ... v2.x와 동일한 의존성 ... */]

  // (v3.0) UI 구현 (순수 HTML/CSS)

            />

              className="submit-button"
