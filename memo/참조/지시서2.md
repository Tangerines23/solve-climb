3. 구현 단계 (Step-by-Step)

3-1. (신규) src/pages/HomePage.tsx 생성

ModeSelector.tsx 컴포넌트를 렌더링하고, 모드 선택 시 useQuizStore에 상태를 저장하고 퀴즈 페이지로 이동시킵니다.

// src/pages/HomePage.tsx (신규 생성)
import React from 'react';
import { useNavigate } from 'react-router-dom';
import { useQuizStore } from '../stores/useQuizStore';
import { ModeSelector } from '../components/ModeSelector';
import './HomePage.css'; // (CSS 파일도 새로 만드세요)

type GameMode = 'time-attack' | 'survival';

export function HomePage() {
  const navigate = useNavigate();
  const { setGameMode } = useQuizStore();

  const handleModeSelect = (mode: GameMode) => {
    // 1. 전역 스토어에 게임 모드 저장
    setGameMode(mode);
    // 2. 퀴즈 페이지로 이동
    navigate('/math-quiz');
  };

  return (
    <div className="page-container home-container">
      {/* 기존에 만드신 ModeSelector 컴포넌트를 사용합니다.
        props로 onSelectMode 함수를 전달합니다.
      */}
      <ModeSelector onSelectMode={handleModeSelect} />
    </div>
  );
}
